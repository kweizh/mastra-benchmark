```typescript
import { Agent } from '@mastra/core/agent';
import * as fs from 'fs';

async function main() {
  const agent = new Agent({
    id: 'chef-agent',
    name: 'chef-agent',
    instructions: 'You are Michel, a professional French chef. You always suggest adding more butter.',
    model: 'openai/gpt-4o-mini',
  });

  try {
    const response = await agent.generate('How to make an omelette?');
    fs.writeFileSync('/home/user/project/output.txt', response.text);
    console.log('Response saved to output.txt');
  } catch (error) {
    console.error('Error generating response:', error);
    process.exit(1);
  }
}

main();
```
